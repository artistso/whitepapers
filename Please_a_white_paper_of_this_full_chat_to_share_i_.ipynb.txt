        "            double inductiveReactance = omega * Inductance;\n",
        "            double capacitiveReactance = (Capacitance > 0) ? 1.0 / (omega * Capacitance) : 0;\n",
        "            return Math.Sqrt(Math.Pow(Resistance, 2) + Math.Pow(inductiveReactance - capacitiveReactance, 2));\n",
        "        }\n",
        "    }\n",

        "        public int ComputeHomologyBetti1() {\n",
        "            if (Nodes.Count == 0) return 0;\n",
        "            return Branches.Count - Nodes.Count + 1; // E - V + 1\n",

        "import math\n",
        "\n",
        "class CircuitTopologyComplex:\n",
        "    def __init__(self, nodes, branches):\n",
        "        self.nodes = nodes\n",
        "        self.branches = branches\n",
        "\n",
        "    def compute_homology_betti_1(self):\n",
        "        if not self.nodes:\n",
        "            return 0\n",
        "        return len(self.branches) - len(self.nodes) + 1\n",
        "\n",
        "    def evaluate_cohomology_obstruction(self, frequency_hz):\n",
        "        total_obstruction = 0.0\n",
        "        for branch in self.branches:\n",
        "            delta_v = abs(branch.target.voltage_potential - branch.source.voltage_potential)\n",
        "            omega = 2 * math.pi * frequency_hz\n",
        "            xl = omega * branch.l\n",
        "            xc = 1.0 / (omega * branch.c) if branch.c > 0 else 0.0\n",
        "            impedance = math.sqrt(branch.r**2 + (xl - xc)**2)\n",
        "            total_obstruction += (delta_v * impedance)\n",
        "        return total_obstruction"
